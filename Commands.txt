#First time Git setup
1. git init
2. git add .
3. git commit -m "any message"
4. git remote add origin https://github.com/Vamsi-767/Property-Management.git
5. git remote set -url origin https://Vamsi-767:TOKEN
6. git push --set upstream origin master
